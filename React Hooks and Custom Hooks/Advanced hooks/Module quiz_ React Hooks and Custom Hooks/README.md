```

  const [restaurantName, setRestaurantName] = useState("Lemon");

  };

  return (
    <div>
      <h1>{restaurantName}</h1>
      <button onClick={updateRestaurantName}>
        Update restaurant name
      </button>
    </div>
  );
};
```
